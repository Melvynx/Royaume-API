import mongoose, { Document } from "mongoose";

// Interface, Schema and Model :

export interface ExampleInterface extends Document {
    date: Date,

    voiceMinute: number,
    messageCount: number,
    memberCount: number
}

export const ExampleSchema = new mongoose.Schema({
    date: {type: Date, default: new Date(new Date().setHours(0, 0, 0, 0))},
    
    voiceMinute: {type: Number, default: 0},
    messageCount: {type: Number, default: 0},
    memberCount: {type: Number, default: 0}
});

const collectionName = "example";
export const ExampleModel = mongoose.model<ExampleInterface>(collectionName, ExampleSchema, collectionName);
